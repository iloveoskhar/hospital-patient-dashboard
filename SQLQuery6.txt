SELECT COUNT(*) AS Total_Patients 
FROM hospital_data;

SELECT AVG(Cost) AS Avg_Cost 
FROM hospital_data;


SELECT Condition, COUNT(*) AS Total_Patients
FROM hospital_data
GROUP BY Condition
ORDER BY Total_Patients DESC;	

SELECT Condition, 
AVG(Cost) AS Avg_Cost
FROM hospital_data
GROUP BY Condition
ORDER BY Avg_Cost DESC;

SELECT AVG(Length_of_Stay) AS
Avg_Length_of_Stay FROM hospital_data;

SELECT Condition, 
AVG(Length_of_Stay) As Avg_Stay
FROM hospital_data
GROUP BY Condition
ORDER BY Avg_Stay DESC;

SELECT Readmission, 
COUNT (*) as Total
FROM hospital_data
GROUP BY Readmission;

SELECT Outcome, 
COUNT (*) as Total
FROM hospital_data
GROUP BY Outcome;

SELECT Cost_Category, 
COUNT (*) as Total
FROM hospital_data
GROUP BY Cost_Category;

SELECT Age_Group, 
COUNT (*) as Total
FROM hospital_data
GROUP BY Age_Group;
